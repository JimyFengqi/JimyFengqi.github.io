    int mod = (int)1e9+7;

    public int[] pathsWithMaxScore(List<String> board) {

                // å¦‚æœã€Œå³è¾¹æ ¼å­ã€åˆæ³•ï¼Œå°è¯•ä»ã€Œå³è¾¹æ ¼å­ã€è¿›è¡Œè½¬ç§»

    }
    
    // äºŒç»´åæ ‡ (x,y) ä¸ idx çš„ç›¸äº’è½¬æ¢

    }
